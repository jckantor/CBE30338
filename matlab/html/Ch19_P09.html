Model

ypred = @(tau1,tau2) 1 - (tau1*exp(-t/tau1) - tau2*exp(-t/tau2))/(tau1-tau2);
err = @(tau1,tau2) (y - ypred(tau1,tau2));
f = @(x) sum(err(x(1),x(2)).^2);
[x,fval] = fminsearch(f,[1 1]);

hold on
plot(t,ypred(x(1),x(2)),'r','LineWidth',2);
hold off
legend('Data','Fitted');
