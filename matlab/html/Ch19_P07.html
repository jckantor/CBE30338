Measurement Data

mB = [92.4; 94.3; 93.8];
mC = [11.1; 10.8; 11.4];

CVX Model

cvx_begin

    % mass balance model
    variables wA wB wC;
    wA + wC == wB;
    wA >= 0;
    wB >= 0;
    wC >= 0;

    % measurement errors
    variables eB(3) eC(3);
    mB == wB + eB;
    mC == wC + eC;

    % least squares objective
    minimize sum(eC.^2 + eB.^2)

cvx_end
